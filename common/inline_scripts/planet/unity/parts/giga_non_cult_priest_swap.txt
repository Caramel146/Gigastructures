
# priests
triggered_planet_modifier = {
    potential = {
        exists = owner
        owner = {
            is_megacorp = no
        }
        $condition$
    }

    job_priest_add = $jobs$
    $multiplier_string$
}
inline_script = {
    script = generic_parts/giga_toggled_code
    code = "
        triggered_desc = {
            trigger = {
                exists = owner
                owner = {
                    is_megacorp = no
                }
                $condition$
            }
            text = job_priest_effect_desc
        }
    "
    toggle = $description$
}
