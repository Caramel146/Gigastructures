
# researchers
triggered_planet_modifier = {
    potential = {
        exists = owner
        owner = {
            is_gestalt = no
        }
        $condition$
    }

    job_researcher_add = $jobs$
    $multiplier_string$
}
inline_script = {
    script = generic_parts/giga_toggled_code
    code = "
        triggered_desc = {
            trigger = {
                exists = owner
                owner = {
                    is_gestalt = no
                }
                $condition$
            }
            text = job_researcher_effect_desc
        }
    "
    toggle = $description$
}
