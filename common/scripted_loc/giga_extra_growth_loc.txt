
defined_text = {
    name = GigaExtraGrowthTooltip

    text = {
        trigger = {
            event_target:giga_growth_situation@root = {
                check_variable = {
                    which = total_growth
                    value > 0
                }
                has_situation_flag = has_species
            }
        }

        localization_key = giga_extra_growth_tooltip_growth
    }

    default = giga_extra_growth_tooltip_no_growth
}
